6. Number of customers not contacted on every day. How many borrowers are not contacted on day 1, day 2 etc.¶

In [23]:
# Get the unique borrowers in the dataset
all_borrowers = set(df["Loan Number"])

# Filter only the completed calls
completed_calls = df[df["Status"] == "completed"]

# Group the completed calls data by call date and get the unique borrowers for each day
completed_borrowers_per_day = completed_calls.groupby(completed_calls["Call time"].dt.date)["Loan Number"].nunique()

# Calculate the number of borrowers not contacted on each day
not_contacted_per_day = {}
for date in completed_borrowers_per_day.index:
    completed_borrowers = completed_borrowers_per_day.loc[date]
    not_contacted_borrowers = len(all_borrowers) - completed_borrowers
    not_contacted_per_day[date] = not_contacted_borrowers

# Create a DataFrame from the results
result_df = pd.DataFrame({"Not Contacted": not_contacted_per_day})

# Print the results as a table
print(result_df)
